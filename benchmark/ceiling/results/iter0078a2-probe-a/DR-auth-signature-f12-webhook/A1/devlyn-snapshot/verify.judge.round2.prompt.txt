```js
// Tiny Express server with a deliberately small API surface.
const crypto = require('crypto');
const express = require('express');
const fs = require('fs');
const path = require('path');

const app = express();

const webhookSecret = fs.readFileSync(path.join(__dirname, '../data/webhook-secret.txt'), 'utf8').trim();
const acceptedEventIds = new Set();

const items = [
  { id: 1, name: 'alpha', qty: 3 },
  { id: 2, name: 'beta', qty: 5 },
];

app.get('/health', (_req, res) => {
  res.json({ status: 'ok' });
});

app.get('/items', (_req, res) => {
  res.json({ items });
});

app.get('/items/:id', (req, res) => {
  const id = Number(req.params.id);
  const item = items.find((it) => it.id === id);
  if (!item) {
    res.status(404).json({ error: 'not_found', id });
    return;
  }
  res.json({ item });
});

app.post('/webhook', express.raw({ type: '*/*' }), (req, res) => {
  const signature = req.get('X-Signature');
  const expectedSignature = crypto
    .createHmac('sha256', webhookSecret)
    .update(req.body)
    .digest('hex');

  if (
    typeof signature !== 'string'
    || !/^[0-9a-f]{64}$/.test(signature)
    || !crypto.timingSafeEqual(Buffer.from(signature, 'hex'), Buffer.from(expectedSignature, 'hex'))
  ) {
    res.status(401).json({ error: 'invalid_signature' });
    return;
  }

  let event;
  try {
    event = JSON.parse(req.body.toString('utf8'));
  } catch {
    res.status(400).json({ error: 'invalid_body' });
    return;
  }

  if (
    typeof event !== 'object' || event === null || Array.isArray(event)
    || typeof event.id !== 'string' || event.id.length === 0
    || typeof event.type !== 'string' || event.type.length === 0
    || typeof event.timestamp !== 'number'
    || typeof event.data !== 'object' || event.data === null || Array.isArray(event.data)
  ) {
    res.status(400).json({ error: 'invalid_body' });
    return;
  }

  if (acceptedEventIds.has(event.id)) {
    res.status(409).json({ error: 'duplicate_event', id: event.id });
    return;
  }

  acceptedEventIds.add(event.id);
  res.json({ accepted: true, id: event.id });
});

if (require.main === module) {
  const port = Number(process.env.PORT) || 3000;
  app.listen(port, () => {
    console.log(`harbor-tools server listening on :${port}`);
  });
}

module.exports = { app };
```
