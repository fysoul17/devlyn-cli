+function fulfillWave(wave) {
+  const warehouses = wave.warehouses.slice().sort((left, right) => (
+    left.distance - right.distance
+    || compareStrings(left.id, right.id)
+    || left.index - right.index
+  ));
+  const orders = wave.orders.slice().sort((left, right) => (
+    right.priority - left.priority
+    || left.submittedAt - right.submittedAt
+    || compareStrings(left.id, right.id)
+  ));
+  const accepted = [];
+  const rejected = [];
+
+  for (const order of orders) {
+    const allocations = [];
+    const deductions = [];
+    let fulfilled = true;
+
+    for (const line of order.lines) {
+      let remaining = line.qty;
+      const candidates = line.singleWarehouse
+        ? warehouses.filter((warehouse) => {
+          let capacity = 0;
+          for (const lot of sortedLotsForSku(warehouse, line.sku)) {
+            capacity = Math.min(line.qty, capacity + lot.qty);
+          }
+          return capacity >= line.qty;
+        })
+        : warehouses;
+
+      for (const warehouse of candidates) {
+        for (const lot of sortedLotsForSku(warehouse, line.sku)) {
+          const qty = Math.min(remaining, lot.qty);
+          if (qty === 0) continue;
+          lot.qty -= qty;
+          deductions.push({ lot, qty });
+          allocations.push({ sku: line.sku, warehouse: warehouse.id, lot: lot.lot, qty });
+          remaining -= qty;
+          if (remaining === 0) break;
+        }
+        if (remaining === 0) break;
+      }
+
+      if (remaining !== 0) {
+        fulfilled = false;
+        break;
+      }
+    }
+
+    if (fulfilled) {
+      accepted.push({ id: order.id, allocations });
+    } else {
+      for (const deduction of deductions) {
+        deduction.lot.qty += deduction.qty;
+      }
+      rejected.push({ id: order.id, reason: 'insufficient_stock', index: order.index });
+    }
+  }

+  const remaining = warehouses

+        sku: lot.sku,

+      || compareStrings(left.sku, right.sku)
+      || compareStrings(left.expires, right.expires)
+      || compareStrings(left.lot, right.lot)
+      || left.warehouseIndex - right.warehouseIndex
+      || left.lotIndex - right.lotIndex

+      .map(({ id, reason }) => ({ id, reason })),

+  };
