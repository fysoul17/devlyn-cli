+function fulfillWave(wave) {
+  const warehouses = wave.warehouses.slice().sort((left, right) => (
+    left.distance - right.distance

+  const rejected = [];

+    let fulfilled = true;

+    for (const line of order.lines) {
+      let remaining = line.qty;
+      const candidates = line.singleWarehouse
+        ? warehouses.filter((warehouse) => {

+          for (const lot of sortedLotsForSku(warehouse, line.sku)) {
+            capacity = Math.min(line.qty, capacity + lot.qty);
+          }
+          return capacity >= line.qty;

+      for (const warehouse of candidates) {
+        for (const lot of sortedLotsForSku(warehouse, line.sku)) {
+          const qty = Math.min(remaining, lot.qty);
+          if (qty === 0) continue;

+          deductions.push({ lot, qty });
+          allocations.push({ sku: line.sku, warehouse: warehouse.id, lot: lot.lot, qty });
+          remaining -= qty;
+          if (remaining === 0) break;
+        }
+        if (remaining === 0) break;
+      }
+
+      if (remaining !== 0) {

+        break;
+      }
+    }
+
+    if (fulfilled) {
+      accepted.push({ id: order.id, allocations });
+    } else {

+      }
